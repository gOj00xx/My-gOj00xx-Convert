-- Flashback
local flashbackActive = false
local flashbackFrames = {}
local maxFrames = 600
local function getCharacter()
    return player.Character or player.CharacterAdded:Wait()
end
local function getHRP(char)
    return char:FindFirstChild("HumanoidRootPart") or char.PrimaryPart
end
local function recordFrame()
    local char = getCharacter()
    local hrp = getHRP(char)
    local hum = char:FindFirstChildWhichIsA("Humanoid")
    if not hrp or not hum then return end
    if #flashbackFrames > maxFrames then
        table.remove(flashbackFrames, 1)
    end
    table.insert(flashbackFrames, {
        CFrame = hrp.CFrame,
        Velocity = hrp.AssemblyLinearVelocity,
        State = hum:GetState(),
        PlatformStand = hum.PlatformStand
    })
end
local function revertFrame()
    local char = getCharacter()
    local hrp = getHRP(char)
    local hum = char:FindFirstChildWhichIsA("Humanoid")
    if not hrp or not hum or #flashbackFrames == 0 then return end
    local lastFrame = flashbackFrames[#flashbackFrames]
    table.remove(flashbackFrames, #flashbackFrames)
    hrp.CFrame = lastFrame.CFrame
    hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    hum:ChangeState(lastFrame.State)
    hum.PlatformStand = lastFrame.PlatformStand
end

RunService.Stepped:Connect(function()
    if flashbackActive then
        revertFrame()
    else
        recordFrame()
    end
end)

local function updateESP(player)
    if player == game.Players.LocalPlayer then return end
    if espEnabled and player.Character then
        local character = player.Character
        local highlight = character:FindFirstChild("Highlight")
        if not highlight then
            highlight = Instance.new("Highlight")
            highlight.Name = "Highlight"
            highlight.Adornee = character
            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            highlight.FillColor = Color3.fromRGB(255, 255, 255)
            highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
            highlight.Parent = character
        end
        createNameTag(character, player.Name)
    elseif player.Character then
        local highlight = player.Character:FindFirstChild("Highlight")
        if highlight then
            highlight:Destroy()
        end
        local nameTag = player.Character:FindFirstChild("ESPNameTag")
        if nameTag then
            nameTag:Destroy()
        end
    end
end

local function clearESP()
    for _, p in pairs(PlayersService:GetPlayers()) do
        if p ~= player and p.Character then
            local highlight = p.Character:FindFirstChild("Highlight")
            if highlight then
                highlight:Destroy()
            end
            local nameTag = p.Character:FindFirstChild("ESPNameTag")
            if nameTag then
                nameTag:Destroy()
            end
        end
    end
end

PlayersTab:CreateToggle({
    Name = "Players Esp",
    CurrentValue = false,
    Callback = function(value)
        espEnabled = value
        if value then
            for _, p in pairs(PlayersService:GetPlayers()) do
                updateESP(p)
                local charConnection
                charConnection = p.CharacterAdded:Connect(function()
                    if espEnabled then
                        updateESP(p)
                    end
                end)
                espConnections[p] = charConnection
            end
            local playerAddedConnection = PlayersService.PlayerAdded:Connect(function(newPlayer)
                if espEnabled then
                    updateESP(newPlayer)
                    local charConnection
                    charConnection = newPlayer.CharacterAdded:Connect(function()
                        if espEnabled then
                            updateESP(newPlayer)
                        end
                    end)
                    espConnections[newPlayer] = charConnection
                end
            end)
            local playerRemovingConnection = PlayersService.PlayerRemoving:Connect(function(leavingPlayer)
                if espConnections[leavingPlayer] then
                    espConnections[leavingPlayer]:Disconnect()
                    espConnections[leavingPlayer] = nil
                end
            end)
            espConnections["PlayerAdded"] = playerAddedConnection
            espConnections["PlayerRemoving"] = playerRemovingConnection
        else
            clearESP()
            for _, conn in pairs(espConnections) do
                conn:Disconnect()
            end
            espConnections = {}
        end
    end
})

                TweenService:Create(element, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.5}):Play()
                uiStrokeTextBox.Color = originalStrokeColor
            end)
        end

        addHoverEffect(textBox)

        local sendButton = Instance.new("TextButton")
        sendButton.Size = UDim2.new(0, 35, 0, 35)
        sendButton.Position = UDim2.new(0.9, -10, 0.5, -17.5)
        sendButton.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        sendButton.BackgroundTransparency = 0.7
        sendButton.Text = "↑"
        sendButton.TextColor3 = Color3.fromRGB(200, 200, 200)
        sendButton.Font = Enum.Font.GothamBold
        sendButton.TextSize = 16
        sendButton.Parent = frame

        local uiCornerButton = Instance.new("UICorner")
        uiCornerButton.CornerRadius = UDim.new(0, 17.5)
        uiCornerButton.Parent = sendButton

        local uiStrokeButton = Instance.new("UIStroke")
        uiStrokeButton.Thickness = 1
        uiStrokeButton.Color = Color3.fromRGB(100, 100, 100)
        uiStrokeButton.Transparency = 0.4
        uiStrokeButton.Parent = sendButton

        local function addButtonHover(element)
            local originalStrokeColor = uiStrokeButton.Color
            
            element.MouseEnter:Connect(function()
                TweenService:Create(element, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.6}):Play()
                uiStrokeButton.Color = Color3.fromRGB(150, 150, 150)
            end)
            
            element.MouseLeave:Connect(function()
                TweenService:Create(element, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.7}):Play()
                uiStrokeButton.Color = originalStrokeColor
            end)
        end

        addButtonHover(sendButton)

        local statusLabel = Instance.new("TextLabel")
        statusLabel.Size = UDim2.new(0.9, 0, 0, 20)
        statusLabel.Position = UDim2.new(0.05, 0, 0.85, 0)
        statusLabel.BackgroundTransparency = 1
        statusLabel.TextColor3 = Color3.fromRGB(150, 150, 150)
        statusLabel.Font = Enum.Font.Gotham
        statusLabel.TextSize = 12
        statusLabel.Text = ""
        statusLabel.TextXAlignment = Enum.TextXAlignment.Left
        statusLabel.Parent = frame

        local function animateOpen()
            local tweenInfo = TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
            local goal = {Position = UDim2.new(0.5, -250, 0.9, -30)}
            TweenService:Create(frame, tweenInfo, goal):Play()
        end

        animateOpen()

        local dragging = false
        local dragInput
        local dragStart
        local startPos

        dragHandle.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = true
                dragStart = input.Position
                startPos = frame.Position
                
                input.Changed:Connect(function()
                    if input.UserInputState == Enum.UserInputState.End then
                        dragging = false
                    end
                end)
            end
        end)
        
        UIS.InputChanged:Connect(function(input)
            if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                local delta = input.Position - dragStart
                frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            end
        end)

        sendButton.MouseButton1Click:Connect(function()
            local input = textBox.Text
            local url, errorMsg = formatUrl(input)
            
            if errorMsg then
                statusLabel.Text = errorMsg
                statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
                wait(3)
                statusLabel.Text = ""
                return
            end
            
            statusLabel.Text = "Opening " .. input .. "..."
            statusLabel.TextColor3 = Color3.fromRGB(100, 200, 100)
            open_browser(url, true)
            
            wait(3)
            statusLabel.Text = ""
        end)

        textBox.FocusLost:Connect(function(enterPressed)
            if enterPressed then
                sendButton.MouseButton1Click:Fire()
            end
        end)
    end
})
